public String toString()
{
    return new StringBuilder(SomeConcept.class.getSimpleName())
        .append('(')
        .append("someProperty=").append('"' + someProperty + '"').append(", ")
        .append("optionalProperty=").append((optionalProperty == null) ? "not present" : String.format("\"%s\"", optionalProperty)).append(", ")
        .append("setProperty=").append(setProperty)
        .append(')')
        .toString();
}
